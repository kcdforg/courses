PHP String Functions

Learn essential functions for working with strings in PHP.

Common String Functions

strlen() - String Length

<?php
    $str = "Hello World";
    echo strlen($str); // Outputs 11
?>

str_word_count() - Count Words

<?php
    $str = "Hello World PHP";
    echo str_word_count($str); // Outputs 3
?>

strrev() - Reverse String

<?php
    $str = "Hello";
    echo strrev($str); // Outputs "olleH"
?>

strpos() - Find Substring Position

<?php
    $str = "Hello World";
    echo strpos($str, "World"); // Outputs 6
    echo strpos($str, "PHP"); // Outputs false
?>

str_replace() - Replace Text

<?php
    $str = "Hello World";
    $new = str_replace("World", "PHP", $str);
    echo $new; // Outputs "Hello PHP"
?>

substr() - Extract Substring

<?php
    $str = "Hello World";
    echo substr($str, 0, 5);   // Outputs "Hello"
    echo substr($str, 6);      // Outputs "World"
    echo substr($str, -5);     // Outputs "World"
?>

strtoupper() / strtolower()

<?php
    $str = "Hello World";
    echo strtoupper($str); // Outputs "HELLO WORLD"
    echo strtolower($str); // Outputs "hello world"
?>

trim() / ltrim() / rtrim()

<?php
    $str = "  Hello World  ";
    
    echo trim($str);   // Removes spaces both sides
    echo ltrim($str);  // Removes left spaces
    echo rtrim($str);  // Removes right spaces
?>

str_split() - Convert String to Array

<?php
    $str = "Hello";
    $arr = str_split($str);
    // Result: ["H", "e", "l", "l", "o"]
    
    $arr2 = str_split($str, 2);
    // Result: ["He", "ll", "o"]
?>

explode() / implode()

<?php
    // Split string into array
    $str = "Apple,Banana,Orange";
    $arr = explode(",", $str);
    
    // Join array into string
    $str2 = implode(",", $arr);
?>

ucfirst() / ucwords()

<?php
    $str = "hello world";
    
    echo ucfirst($str);  // Outputs "Hello world"
    echo ucwords($str);  // Outputs "Hello World"
?>

number_format() - Format Numbers

<?php
    $num = 1234.56789;
    echo number_format($num, 2); // Outputs "1,234.57"
?>
