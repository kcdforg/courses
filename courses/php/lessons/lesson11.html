PHP Array Functions

Learn useful built-in functions for working with arrays.

Common Array Functions

count() - Get Array Length

<?php
    $fruits = ["Apple", "Banana", "Orange"];
    echo count($fruits); // Outputs 3
?>

array_push() - Add Elements

<?php
    $fruits = ["Apple", "Banana"];
    array_push($fruits, "Orange", "Mango");
    // $fruits now contains 4 elements
?>

array_pop() - Remove Last Element

<?php
    $fruits = ["Apple", "Banana", "Orange"];
    $last = array_pop($fruits);
    echo $last; // Outputs "Orange"
?>

array_shift() / array_unshift()

<?php
    $fruits = ["Apple", "Banana", "Orange"];
    
    // Remove first element
    $first = array_shift($fruits);
    
    // Add element at beginning
    array_unshift($fruits, "Mango");
?>

in_array() - Check if Value Exists

<?php
    $fruits = ["Apple", "Banana", "Orange"];
    
    if (in_array("Banana", $fruits)) {
        echo "Found!";
    }
?>

array_search() - Find Key of Value

<?php
    $fruits = ["Apple", "Banana", "Orange"];
    $key = array_search("Banana", $fruits);
    echo $key; // Outputs 1
?>

array_keys() / array_values()

<?php
    $person = ["name" => "John", "age" => 25, "city" => "NYC"];
    
    // Get all keys
    $keys = array_keys($person);
    
    // Get all values
    $values = array_values($person);
?>

sort() / rsort() / asort() / arsort()

<?php
    $fruits = ["Orange", "Apple", "Banana"];
    
    sort($fruits);       // Ascending order
    rsort($fruits);      // Descending order
    asort($fruits);      // Ascending with keys
    arsort($fruits);     // Descending with keys
?>

array_merge() - Combine Arrays

<?php
    $arr1 = ["a" => "red", "b" => "green"];
    $arr2 = ["c" => "blue", "d" => "yellow"];
    
    $result = array_merge($arr1, $arr2);
?>

array_slice() / array_splice()

<?php
    $fruits = ["Apple", "Banana", "Orange", "Mango"];
    
    // Get portion of array
    $slice = array_slice($fruits, 1, 2); // ["Banana", "Orange"]
    
    // Replace elements
    array_splice($fruits, 1, 2, ["Grape", "Kiwi"]);
?>

array_filter() / array_map()

<?php
    $numbers = [1, 2, 3, 4, 5];
    
    // Filter even numbers
    $even = array_filter($numbers, function($n) {
        return $n % 2 == 0;
    });
    
    // Double each number
    $doubled = array_map(function($n) {
        return $n * 2;
    }, $numbers);
?>
